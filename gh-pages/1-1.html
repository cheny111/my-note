    var fs = api.require('fs');
        fs.exist({
            path: 'fs://ams_area.json'
        }, function(ret, err) {
            if (ret.exist) {
                //1有 0无 2失败
                $api.byId("ams_area").value=1;
            }else{
              api.showProgress({
                title: '初始化数据中...',
                text: '先喝杯茶...',
                modal: false
              });
              //执行下载
              api.ajax({
                  url: SERVER_URL+'/index.php/Api/Update/GetData',
                  method: 'post',
                  timeout:settime,
                  headers:{
                        "X-APICloud-AppId" : appID,
                        "X-APICloud-AppKey" : appKey
                    },
                  data: {
                      values: {
                          "user_id": id, "t": 1

                      }
                  }
              }, function(ret, err) {
                  if (ret) {
                  //api.alert({ msg: JSON.stringify(ret) });
                    if(ret.flag==0){
                      api.toast({
                          msg: ret.msg,
                          duration: 2000,
                          location: 'bottom'
                      });
                    }else {
                        var downloadurl= SERVER_URL+ret.body[0]['url'];
                        var updatetime=ret.body[0]['updatetime'];
                        api.download({
                            url:downloadurl,
                            savePath: 'fs://ams_area.json',
                            report: true,
                            cache: false,
                            allowResume: true
                        }, function(ret, err) {
                          //api.alert({ msg: JSON.stringify(ret) });
                      if (ret.state == 1) {
                             //1有 0无 2失败
                          $api.byId("ams_area").value=1;

                          up(1,updatetime)


                          judge()
                      } else if(ret.state == 0){
                          //正在下载当中

                      }else{
                          //1有 0无 2失败
                          $api.byId("ams_area").value=2;
                          judge()
                      }
                    });
                }
              }
            });
          }
        });
